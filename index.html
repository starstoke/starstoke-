// Currency selector (без изменений) const currencyButtons = document.querySelectorAll('.currency-btn'); const prices = document.querySelectorAll('.card-price .currency'); currencyButtons.forEach(button => { button.addEventListener('click', () => { currencyButtons.forEach(btn => btn.classList.remove('active')); button.classList.add('active'); const currency = button.dataset.currency; prices.forEach(price => { price.textContent = currency; }); }); }); // Contact form submission (улучшено для плавности: добавлена анимация opacity) const contactForm = document.getElementById('contact-form'); if (contactForm) { contactForm.style.opacity = '1'; // Плавное появление при загрузке contactForm.addEventListener('submit', function(event) { event.preventDefault(); const form = event.target; fetch(form.action, { method: 'POST', body: new FormData(form), headers: { 'Accept': 'application/json' } }).then(response => { if (response.ok) { alert('Message sent successfully!'); form.reset(); contactForm.style.opacity = '0.5'; // Краткая анимация сброса setTimeout(() => { contactForm.style.opacity = '1'; }, 300); } else { alert('Error sending message. Please try again.'); } }).catch(error => { alert('Error sending message. Please try again.'); }); }); } });
