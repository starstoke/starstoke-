function showArtistItem(index) { index = (index + totalArtists) % totalArtists; const isMobile = window.innerWidth <= 768; const translateValue = isMobile ? -index * 100 : -index * (100 / 3); artistGrid.style.transform = translateX(${translateValue}%); currentArtistIndex = index; } artistControls.forEach(control => { control.addEventListener('click', () => { if (control.classList.contains('prev')) { showArtistItem(currentArtistIndex - 1); } else if (control.classList.contains('next')) { showArtistItem(currentArtistIndex + 1); } }); }); setInterval(() => { showArtistItem(currentArtistIndex + 1); }, 5000); // Handle window resize for artist carousel (с debounce для плавности в мобильке) let resizeTimeout; window.addEventListener('resize', () => { clearTimeout(resizeTimeout); resizeTimeout = setTimeout(() => { showArtistItem(currentArtistIndex); }, 100); }); // Currency selector const currencyButtons = document.querySelectorAll('.currency-btn'); const prices = document.querySelectorAll('.card-price .currency'); currencyButtons.forEach(button => { button.addEventListener('click', () => { currencyButtons.forEach(btn => btn.classList.remove('active')); button.classList.add('active'); const currency = button.dataset.currency; prices.forEach(price => { price.textContent = currency; }); }); }); // Contact form submission (улучшено для плавности) const contactForm = document.getElementById('contact-form'); if (contactForm) { contactForm.style.opacity = '1'; // Плавное появление contactForm.addEventListener('submit', function(event) { event.preventDefault(); const form = event.target; fetch(form.action, { method: 'POST', body: new FormData(form), headers: { 'Accept': 'application/json' } }).then(response => { if (response.ok) { alert('Message sent successfully!'); form.reset(); contactForm.style.opacity = '0.5'; // Краткая анимация setTimeout(() => { contactForm.style.opacity = '1'; }, 300); } else { alert('Error sending message. Please try again.'); } }).catch(error => { alert('Error sending message. Please try again.'); }); }); } });
